# ============================================================
# STEP 1 — REGISTER & LOGIN
# ============================================================

# Register tradie
curl -s -X POST http://localhost:5000/api/users/register \
  -H "Content-Type: application/json" \
  -d "{\"name\":\"Test Tradie\",\"email\":\"tradie@test.com\",\"password\":\"password123\",\"role\":\"tradie\"}"

# Register regular user
curl -s -X POST http://localhost:5000/api/users/register \
  -H "Content-Type: application/json" \
  -d "{\"name\":\"S. Admin\",\"email\":\"super@admin.com\",\"password\":\"password.admin.123\",\"role\":\"user\"}"

# Login as tradie — copy the token from the response
curl -s -X POST http://localhost:5000/api/users/login \
  -H "Content-Type: application/json" \
  -d "{\"email\":\"tradie@test.com\",\"password\":\"password123\"}"

# Login as admin — copy the token from the response
curl -s -X POST http://localhost:5000/api/users/login \
  -H "Content-Type: application/json" \
  -d "{\"email\":\"super@admin.com\",\"password\":\"password.admin.123\"}"
